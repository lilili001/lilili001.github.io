    http://localhost:3000/webpack-dev-server/index.html

  • 通过 –inline –hot

    $ webpack-dev-server --content-base src --port=3000 --inline --hot
  • 通过 npm run dev

将这句命令 $ webpack-dev-server --content-base src --port=3000 --inline --hot

package.json:
"dev":"./node_modules/.bin/webpack-dev-server --content-base src --port=3000 --inline --hot"

dev-server的配置

devserver的配置选项 功能描述
contentBase 默认webpack-dev-server会为根文件夹提供本地服务器，如果想为另外一个目录下的文件提供本地服务器，应该在这里设置其所在目录（本例设置到“public”目录）,个人认为是 html所在目录
port 设置默认监听端口，如果省略，默认为”8080“
inline 设置为true，当源文件改变时会自动刷新页面
historyApiFallback 在开发单页应用时非常有用，它依赖于HTML5 history API，如果设置为true，所有的跳转将指向index.html

把这些命令加到webpack的配置文件中，现在的配置文件webpack.config.js如下所示

module.exports = {
  devtool: 'eval-source-map',

  entry:  __dirname + "/app/main.js",
  output: {
    path: __dirname + "/public",
    filename: "bundle.js"
  },

  devServer: {
    contentBase: "./public",//本地服务器所加载的页面所在的目录
    historyApiFallback: true,//不跳转
    inline: true//实时刷新
  } 
}

let , const

let,const :block scope 块级作用域 作用域 {} 内，{}外是无法访问的

if(true){
    let name = "alice";
}
console.log(name) // nam is not defined

name只是作用在{} 内,但是var 对{}而言是直接暴露在外面的,只是在函数内有作用域的概念

区别：

  • var 可以在全局中重复声明 这样的结果是会覆盖别人写的变量

  • let 和 const 不能在同一个作用域中重复声明变量

    let name = "alice";
let name = "joe";
console.log(name); //Identifier 'name' has already been declared

//不同作用域 虽然变量命一样 但是值不一样

console.log(name);  //joe
if(true){
   let name = "melody";
   console.log(name)//melody
}

  • let 可以重新赋值， 但是const不可以

    let name = "alice";
    name="sdf";  //不报错

const age# "AGE";
     age = "arg";// 报错 Assignment to constant variable.

  • const 属性可以赋值

当多个ajax同时执行的时候,无法预知哪个ajax先返回,哪个后返回 返回的顺序是不定的

promise相当于一个承诺,代表一定会给你一个反馈,promise 可返回promise

let username;

 const userPromise = axios.get('https://api.github.com/users');

 userPromise
    .then( response => {

        //返回promise
        return axios.get(`https://api.github.com/users/${username}/repos`);
    })
    //第二个promise的返回

const repos = [

    { id:2,name:'jsawesome',owner:'vanpelt' },

    {name:'mojombo',location:'San Fransisco'},
    {name:'vanpelt',location:'Bay Minette'},
    {name:'wycats',location:'Los Angeles CA'}
 ]

 function getRepoById(id){
    return new Promise( (resolve,reject)=>{
        const repo = repos.find( repo => repo.id === id );
        repo ? resolve(repo) : reject(Error('No repo found'));
    });
 }

 function comboundOwner(repo){
    return new Promise((resolve,reject)=>{
        const owner = owners.find( owner => owner.name === repo.owner );
        if(owner){
            repo.owner = owner;
            resolve(repo);
        }else{
            reject(Error('can not found the owner'))
        }
    })
 }

 //得到repo后 返回 new Promise ( comboundOwner )
 getRepoById(1)
    .then(
        repo => {
            //console.log(repo);//{id: 1, name: "grit", owner: "mojombo"}
            return comboundOwner(repo);//返回promise repo
